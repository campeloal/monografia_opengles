varying float intensity;
	
void main()
{
	vec4 color;
	
	if (intensity > 0.95)
		color = vec4(0.2,0.2,0.2,1.0);
	else if (intensity > 0.5)
		color = vec4(0.21,0.21,0.21,1.0);
	else if (intensity > 0.25)
		color = vec4(0.57,0.56,0.56,1.0);
	else
		color = vec4(0.84,0.83,0.83,1.0);

	gl_FragColor = color;
}
